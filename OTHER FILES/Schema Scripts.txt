CREATE DATABASE charaterCreator;

CREATE TABLE UserInfo (
	userID int IDENTITY(1,1) AUTO_INCREMENT,
	username VARCHAR(20) NOT NULL,
	password VARCHAR(20) NOT NULL,
	UNIQUE (username),
	PRIMARY KEY (userID)
)

CREATE TABLE Character(
	charID INT IDENTITY(1,1) AUTO_INCREMENT,
	name VARCHAR(50) NOT NULL,
	player VARCHAR(20) NOT NULL,
	classID INT NOT NULL,
	subclassID INT,
	level INT NOT NULL,
	raceID INT NOT NULL,
	experience INT NOT NULL, 
	alignment VARCHAR(2) NOT NULL,
	hitDie VARCHAR(10) NOT NULL,
	hitPointsOne VARCHAR(30) NOT NULL,
	hitPointsHigher VARCHAR(30) NOT NULL,
	
	PRIMARY Key (charNum)
	FOREIGN KEY (player) REFERENCES UserInfo(userID)
		ON DELETE CASCADE
		ON UPDATE CASCADE
	FOREIGN KEY (classID) REFERENCES Class(classID)
		ON DELETE CASCADE
		ON UPDATE CASCADE
	FOREIGN KEY (subclassID) REFERENCES SubClass(subclassID)
		ON DELETE CASCADE
		ON UPDATE CASCADE
	FOREIGN KEY (raceID) REFERENCES Race(raceID)
		ON DELETE CASCADE
		ON UPDATE CASCADE

)

CREATE TABLE Spells(
	spellID INT IDENTITY(1,1) AUTO_INCREMENT,
	spellName VARCHAR(30) NOT NULL,
	spellsDesription VARCHAR(600) NOT NULL,
	spellLevel INT NOT NULL,
	spellClass VARCHAR(30) NOT NULL
)

CREATE TABLE Skills(
	ability VARCHAR(30) NOT NULL,
	skillID INT IDENTITY(1,1) AUTO_INCREMENT,
	skillName VARCHAR(30) NOT NULL,
	skillDescription VARCHAR(30) NOT NULL
)

CREATE TABLE Equipment(
	equipmentID INT IDENTITY(1,1) AUTO_INCREMENT,
	equipmentName VARCHAR(30) NOT NULL,
	equipmentDescription VARCHAR(600) NOT NULL,
	gold INT NOT NULL
)

CREATE TABLE Leveling(
	levelingID INT IDENTITY(1,1) AUTO_INCREMENT,
	level INT NOT NULL,
	experience INT NOT NULL,
	ProficiencyBonus INT NOT NULL
)

CREATE TABLE Races(
	raceID INT IDENTITY(1,1) AUTO_INCREMENT,
	raceName VARCHAR(30) NOT NULL,
	abilityScoreIncrease VARCHAR(600) NOT NULL,
	raceDescription VARCHAR(600) NOT NULL
)

CREATE TABLE Languages(
	languageID INT IDENTITY(1,1) AUTO_INCREMENT,
	languageName VARCHAR(30),
	languageDescription VARCHAR(600) NOT NULL
)
